78/// @brief Contains data about a block tree level under construction
79template <std::signed_integral size_type, typename RankSelect>
80struct LevelData {
81 /// @brief Contains a 1 for each internal block (= block with children)
82 /// and a 0 for each block that has a back pointer
83 std::unique_ptr<pasta::BitVector> is_internal;
84 /// @brief Rank data structure for is_internal
85 std::unique_ptr<RankSelect> is_internal_rank;
86 /// @brief The block from which a back block is copying
87 std::unique_ptr<std::vector<size_type>> pointers;
88 /// @brief The offset into the block from which the back block is copying
89 std::unique_ptr<std::vector<size_type>> offsets;
90 /// @brief The number of back blocks pointing to the block
91 std::unique_ptr<std::vector<size_type>> counters;
92 /// @brief Block start indices
93 std::unique_ptr<std::vector<size_type>> block_starts;
94 /// @brief The block size on this level
95 int64_t block_size;
96 /// @brief The index of the current level.
97 /// First level is 0, second level is 1 etc.
98 int64_t level_index;
99 /// @brief The number of blocks on the current level
100 int64_t num_blocks;
101
102 LevelData(const int64_t level_index_,
103 const int64_t block_size_,
104 const int64_t num_blocks_)
105 : is_internal(nullptr),
106 is_internal_rank(nullptr),
107 pointers(new std::vector<size_type>()),
108 offsets(new std::vector<size_type>()),
109 counters(new std::vector<size_type>()),
110 block_starts(new std::vector<size_type>()),
111 block_size(block_size_),
112 level_index(level_index_),
113 num_blocks(num_blocks_) {}
114
115 /// @brief Checks whether a block is adjacent in the text
116 /// to its successor on this level
117 [[nodiscard]] bool next_is_adjacent(size_t i) const {
118 return (*block_starts)[i] + block_size == (*block_starts)[i + 1];
119 }
120};

122/// @brief Contains data about the occurrences of a hashed block pair
123template <std::signed_integral size_type>
124struct PairOccurrences {
125 /// @brief The first block in the text in which the content appears
126 size_type first_occ_block;
127 /// @brief A list of block indices in which the content of the hashed block
128 /// pair appears
129 ///
130 /// We're using an std::list here instead of an std::vector, since the
131 /// reallocation upon insertion lead to issues during parallel access, when
132 /// another thread tries to access the vector during reallocation.
133 std::vector<size_type> occurrences;
134
135 /// @brief Initialize the occurrences of a hashed block pair.
136 ///
137 /// Note, that this only sets the first occurrence to the given block index,
138 /// but does not add it to the occurrences list.
139 /// @param first_occ_block_ The block index of the pair's first block.
140 inline explicit PairOccurrences(size_type first_occ_block_)
141 : first_occ_block(first_occ_block_),
142 occurrences() {}
143
144 PairOccurrences(PairOccurrences&&) noexcept = default;
145 PairOccurrences& operator=(PairOccurrences&&) = default;
146
147 /// @brief Add a block index to the occurrences.
148 /// @param block_index The block index to add to the occurrences.
149 void add_block_pair(size_type block_index) {
150 occurrences.push_back(block_index);
151 }
152
153 /// @brief If the given block index is an earlier occurrence, update it
154 /// @param block_index The block index of an occurrence
155 void update(size_type block_index) {
156 first_occ_block = std::min<size_type>(first_occ_block, block_index);
157 }
158};

160/// @brief Contains data about the occurrences of a hashed block
161template <std::signed_integral size_type>
162struct BlockOccurrences {
163 /// @brief Represents the first occurrence of a block
164 struct [[gnu::packed]] FirstOccurrence {
165 /// @brief Block index of the first occurrence of the block's content
166 int64_t block : 40;
167 /// @brief The offset into the block at which that first occurrence occurs
168 int32_t offset : 24;
169
170 inline FirstOccurrence(int64_t first_occ_block_,
171 int32_t first_occ_offset_)
172 : block(first_occ_block_),
173 offset(first_occ_offset_) {}
174 };
175
176 static_assert(std::atomic<FirstOccurrence>::is_always_lock_free,
177 "first occurrence must be able to be atomically updated");
178 static_assert(sizeof(FirstOccurrence) == 8,
179 "should be size of computer word");
180
181 // @brief The block index and offset of the first occurrence of this block's
182 // content
183 std::atomic<FirstOccurrence> first_occ;
184
185 /// @brief A list of block indices in which the content of the hashed block
186 /// occurs
187 std::vector<size_type> occurrences;
188
189 /// @brief Initialize the occurrences of a hashed block.
190 ///
191 /// Note, that this only sets the first occurrence to the given block index,
192 /// but does not add it to the occurrences list.
193 /// @param first_occ_block_ The block index of the block's first occurrence.
194 explicit BlockOccurrences(size_type first_occ_block_)
195 : first_occ({first_occ_block_, 0}),
196 occurrences() {}
197
198 BlockOccurrences(const BlockOccurrences& other)
199 : first_occ(other.first_occ.load()),
200 occurrences(other.occurrences) {}
201
202 BlockOccurrences(BlockOccurrences&& other) noexcept
203 : first_occ(other.first_occ.load()),
204 occurrences(std::move(other.occurrences)) {}
205
206 ~BlockOccurrences() = default;
207
208 BlockOccurrences& operator=(BlockOccurrences&& other) noexcept {
209 first_occ = other.first_occ.load();
210 occurrences = std::move(other.occurrences);
211 return *this;
212 }
213
214 /// @brief Add a block index to the occurrences.
215 /// @param block_index The block index to add to the occurrences.
216 void add_block(size_type block_index) {
217 occurrences.push_back(block_index);
218 }
219
220 /// @brief If the given block index and offset are an earlier occurrence,
221 /// update them
222 /// @param block_index The block index of an occurrence
223 /// @param block_offset The offset of that occurrence
224 void update(size_type block_index, size_type block_offset) {
225 FirstOccurrence prev_first_occ = this->first_occ.load();
226 FirstOccurrence set(block_index, block_offset);
227 while (block_index < prev_first_occ.block &&
228 !first_occ.compare_exchange_weak(prev_first_occ, set)) {
229 }
230 }
231};

233/// @brief An update function for the sharded hash map that updates the
234/// occurrences of a hashed block pair
235template <std::integral input_type, std::signed_integral size_type>
236struct UpdatePairOccurrences {
237 /// @brief The block index to add to the occurrences
238 using InputValue = size_type;
239 /// @brief Update the occurrences of a hashed block pair by adding the new
240 /// block index and updating the first occurrence if needed
241 /// @param occurrences A reference to the occurrences in the map
242 /// @param input_value The new block index to add to the occurrences
243 inline static void update(const MersenneHash<input_type>&,
244 PairOccurrences<size_type>& occurrences,
245 InputValue&& input_value) {
246 occurrences.add_block_pair(input_value);
247 occurrences.update(input_value);
248 }
249
250 /// @brief Initialize the occurrences of a hashed block pair
251 /// @param input_value The block index of the pair's first block
252 /// @return The initialized occurrences only containing the given block pair
253 inline static PairOccurrences<size_type> init(const MersenneHash<input_type>&,
254 InputValue&& input_value) {
255 PairOccurrences occurrences(input_value);
256 occurrences.add_block_pair(input_value);
257 occurrences.update(input_value);
258 return occurrences;
259 }

295/// @brief A mixing functions to provide better avalanching to intermediate hash
296/// values.
297///
298/// https://zimbry.blogspot.com/2011/09/better-bit-mixing-improving-on.html
299constexpr uint64_t mix_select(uint64_t key) {
300 key ^= (key >> 31);
301 key *= 0x7fb5d329728ea185;
302 key ^= (key >> 27);
303 key *= 0x81dadef4bc2dd44d;
304 key ^= (key >> 33);
305 return key;
306}
