39template <std::integral input_type, std::signed_integral size_type>
40class BlockTreeFP2 : public BlockTree<input_type, size_type> {
41 using BitVector = pasta::BitVector;
42 using Rank = pasta::RankSelect<pasta::OptimizedFor::ONE_QUERIES>;
43
44 template <typename key_type,
45 typename value_type,
46 typename hash_type = std::hash<key_type>>
47 using HashMap = ankerl::unordered_dense::map<key_type, value_type, hash_type>;
48
49 using RabinKarp = MersenneRabinKarp<input_type,
50 size_type,
51 internal::sharded::PRIME_EXPONENT>;
52 using RabinKarpHash = MersenneHash<input_type>;
53
54 template <typename value_type>
55 using RabinKarpMap = HashMap<RabinKarpHash, value_type>;
56
57 using LevelData = internal::sharded::LevelData<size_type, Rank>;

59 void construct(const std::vector<input_type>& text) {
60 const size_type text_len = text.size();
61 /// The number of characters a block tree with s top-level blocks and arity
62 /// of strictly tau would exceed over the text size
63 int64_t padding;
64 /// The height of the tree
65 int64_t tree_height;
66 /// The size of the largest blocks (i.e. the top level blocks)
67 int64_t top_block_size;
68
69 this->calculate_padding(padding, text_len, tree_height, top_block_size);
70
71 const bool is_padded = padding > 0;
72
73 std::vector<LevelData> levels;
74
75 // Prepare the top level
76 levels.emplace_back(0, top_block_size, text_len / top_block_size);
77 LevelData& top_level = levels.back();
78 top_level.block_starts->reserve(
79 internal::sharded::ceil_div(text_len, top_level.block_size));
80 for (size_type i = 0; i < text_len; i += top_level.block_size) {
81 top_level.block_starts->push_back(i);
82 }
83 top_level.block_size = top_block_size;
84 top_level.num_blocks = top_level.block_starts->size();
85
86 // Construct the pre-pruned tree level by level
87 for (size_t level = 0; level < static_cast<size_t>(tree_height); level++) {
88 LevelData& current = levels.back();
89 scan_block_pairs(text, current, is_padded);
90 scan_blocks(text, current, is_padded);
91
92 // Generate the next level (if we're not at the last level)
93 if (level < static_cast<size_t>(tree_height) - 1 &&
94 levels.back().block_size > this->max_leaf_length_ * this->tau_) {
95 levels.push_back(std::move(generate_next_level(text, current)));
96 }
97 }
98
99 prune(levels);
100 make_tree(text, levels, padding);
101 }

103 /// @brief Scan through the blocks pairwise in order to identify which blocks
104 /// should
105 /// be replaced with back blocks.
106 ///
107 /// @param text The input string.
108 /// @param level The data for the current level.
109 ///
110 /// @return The block start indices for the next level of the tree
111 static void scan_block_pairs(const std::vector<input_type>& text,
112 LevelData& level,
113 const bool is_padded) {
114 const size_t block_size = level.block_size;
115 const size_t num_blocks = level.num_blocks;
116 const size_t pair_size = 2 * block_size;
117
118 if (num_blocks < 4) {
119 level.is_internal = std::make_unique<BitVector>(num_blocks, true);
120 level.is_internal_rank = std::make_unique<Rank>(*level.is_internal);
121 return;
122 }
123
124 // A map containing hashed block pairs mapped to their indices of the
125 // pairs' first block respectively
126 RabinKarpMap<std::vector<size_type>> map(num_blocks - 1);
127
128 // Set up the packed array holding the markings for each block.
129 // Each mark is a 2-bit number.
130 // The MSB is 1 iff the block and its successor have a prior occurrence.
131 // The LSB is 1 iff the block and its predecessor have a prior occurrence.
132 sdsl::int_vector<2> markings(num_blocks, 0);
133
134 {
135 RabinKarp rk(text,
136 internal::sharded::SIGMA,
137 0,
138 pair_size,
139 internal::sharded::PRIME);
140 for (size_t i = 0; i < num_blocks - 1 - is_padded; ++i) {
141 // If the next block is not adjacent, we cannot hash the pair starting
142 // at the current block
143 if (!level.next_is_adjacent(i)) {
144 continue;
145 }
146 // Move the hasher to the current block pair
147 rk.restart((*level.block_starts)[i]);
148 RabinKarpHash hash = rk.current_hash();
149 map[hash].push_back(i);
150 }
151 }
152
153 // Hash every window and determine for all block pairs whether they have
154 // previous occurrences.
155 RabinKarp rk(text,
156 internal::sharded::SIGMA,
157 0,
158 pair_size,
159 internal::sharded::PRIME);
160 for (size_t i = 0; i < num_blocks - 1 - is_padded; ++i) {
161 if (!level.next_is_adjacent(i) | !level.next_is_adjacent(i + 1)) {
162 continue;
163 }
164 scan_windows_in_block_pair(rk, map, markings, block_size);
165 }
166
167 // Generate the bit vector indicating which blocks are internal
168 level.is_internal = std::make_unique<BitVector>(num_blocks);
169 auto& is_internal = *level.is_internal;
170 is_internal[0] = true;
171 is_internal[num_blocks - 1] = markings[num_blocks - 1] != 0b01;
172 for (size_t i = 0; i < num_blocks - 1; ++i) {
173 const bool block_is_internal = markings[i] != 0b11;
174 is_internal[i] = block_is_internal;
175 }
176 level.is_internal_rank = std::make_unique<Rank>(is_internal);
177 }

179 /// @brief Scan through the windows starting in a block and mark them
180 /// accordingly if they represent the earliest occurrence of some block
181 /// hash.
182 ///
183 /// The supplied `RabinKarp` hasher must be at the start of the block.
184 /// @param rk A Rabin-Karp hasher whose state is at the start of the block.
185 /// @param map The map containing the hashes of block pairs mapped to their
186 /// index.
187 /// @param markings A vector storing the marks on a block. Marks are 2-bit
188 /// integers.
189 /// If the MSB is set, that means that the content of the block and its
190 /// successor has an earlier occurrence. If the LSB being set means that
191 /// the content of the block and its predecessor has an earlier occurrence.
192 /// @param block_size The size of blocks on the current level.
193 static inline void
194 scan_windows_in_block_pair(RabinKarp& rk,
195 RabinKarpMap<std::vector<size_type>>& map,
196 sdsl::int_vector<2>& markings,
197 const size_t block_size) {
198 for (size_t offset = 0; offset < block_size; ++offset, rk.next()) {
199 RabinKarpHash current_hash = rk.current_hash();
200 // Find the hash of the current window among the hashed block pairs.
201 auto found_hash_ptr = map.find(current_hash);
202 if (found_hash_ptr == map.end()) {
203 continue;
204 }
205 auto& [block_pair_hash, block_indices] = *found_hash_ptr;
206
207 // If the hash we found is just the first block pair in the hash map,
208 // then the first block pair has no earlier occurrence.
209 // So we want to skip that one
210 bool skip_first = current_hash.start_ == block_pair_hash.start_;
211
212 // For all pairs with an earlier occurrence,
213 for (size_t i = skip_first; i < block_indices.size(); i++) {
214 auto block_index = block_indices[i];
215 markings[block_index] = markings[block_index] | 0b10;
216 markings[block_index + 1] = markings[block_index + 1] | 0b01;
217 }
218 map.erase(found_hash_ptr);
219 }
220 }

222 /// @brief Determine the positions for each block's earliest occurrence if
223 /// there is any.
224 ///
225 /// @param s The input text
226 /// @param level_data The data for the current level
227 /// @param is_padded true, iff the last block of the level extends past the
228 /// end of the text
229 static void scan_blocks(const std::vector<input_type>& s,
230 LevelData& level_data,
231 const bool is_padded) {
232 const size_t block_size = level_data.block_size;
233 const size_t num_blocks = level_data.num_blocks;
234 const std::vector<size_type>& block_starts = *level_data.block_starts;
235
236 level_data.pointers = std::make_unique<std::vector<size_type>>(
237 num_blocks,
238 internal::sharded::NO_EARLIER_OCC);
239 level_data.offsets =
240 std::make_unique<std::vector<size_type>>(num_blocks, 0);
241 level_data.counters =
242 std::make_unique<std::vector<size_type>>(num_blocks, 0);
243
244 if (num_blocks <= 2) {
245 return;
246 }
247
248 // A map with hashed slices as keys, which map to a vector of links,
249 // describing a link between a (potential) back block to their source block.
250 // In addition to the vector, there is a boolean which denotes whether a
251 // hash has already been processed
252 RabinKarpMap<std::vector<size_type>> links(num_blocks);
253 for (size_t i = 0; i < num_blocks - is_padded; ++i) {
254 const RabinKarpHash hash = RabinKarp(s,
255 internal::sharded::SIGMA,
256 block_starts[i],
257 block_size,
258 internal::sharded::PRIME)
259 .current_hash();
260 links[hash].push_back(i);
261 }
262
263 // Hash every window and find the first occurrences for every block.
264 RabinKarp rk(s,
265 internal::sharded::SIGMA,
266 block_starts[0],
267 block_size,
268 internal::sharded::PRIME);
269 for (size_t current_block_index = 0;
270 current_block_index < num_blocks - is_padded - 1;
271 ++current_block_index) {
272 if (static_cast<int64_t>(rk.init_) != block_starts[current_block_index]) {
273 rk.restart(block_starts[current_block_index]);
274 }
275
276 if (level_data.next_is_adjacent(current_block_index)) {
277 scan_windows_in_block(rk, links, level_data, current_block_index);
278 continue;
279 }
280 }
281 }
282 /// @brief Scans through block-sized windows starting inside one block and
283 /// tries to find earlier occurrences of blocks. Non-internal blocks will
284 /// have their respective m_source_blocks and m_offsets entries populated.
285 /// @param rk A Rabin-Karp hasher whose current state is at a block start.
286 /// @param links A map whose keys are hashed blocks and the values
287 /// are all block indices of blocks matching the hash in ascending order.
288 /// @param current_block_internal_index The index of the block which the
289 /// Rabin-Karp hasher is situated in only with respect to *internal blocks*
290 /// on the current level, disregarding back blocks.
291 /// @param num_hashes The number of times the Rabin-Karp hasher should hash.
292 static void scan_windows_in_block(RabinKarp& rk,
293 RabinKarpMap<std::vector<size_type>>& links,
294 LevelData& level_data,
295 const size_type current_block_index) {
296 const BitVector& is_internal = *level_data.is_internal;
297 for (size_type offset = 0; offset < level_data.block_size;
298 ++offset, rk.next()) {
299 const RabinKarpHash hash = rk.current_hash();
300 // Find all blocks in the multimap that match our hash
301 auto found = links.find(hash);
302 if (found == links.end()) {
303 continue;
304 }
305 const auto& [block_hash, found_blocks] = *found;
306 const size_t num_found_blocks = found_blocks.size();
307 // In this case, we are hashing an actual block right now (not just an
308 // arbitrary window). As a result, the first block in the vector is the
309 // block we are currently hashing in
310 for (size_t i = 0; i < num_found_blocks; ++i) {
311 const size_type block_index = found_blocks[i];
312 if (block_index == current_block_index ||
313 (offset > 0 && block_index == current_block_index + 1)) {
314 continue;
315 }
316 (*level_data.pointers)[block_index] = current_block_index;
317 (*level_data.offsets)[block_index] = offset;
318 // We increment the counter for the block that is being pointed to
319 // if the current block is actually a back block
320 // If the offset is greater than 0,
321 // then it also overlaps into the next block
322 const bool is_back_block = !is_internal[block_index];
323 (*level_data.counters)[current_block_index] += is_back_block;
324 (*level_data.counters)[current_block_index + 1] +=
325 is_back_block && (offset > 0);
326 }
327 links.erase(found);
328 }
329 }

331 /// @brief Generate the block size, number of block and block start indices
332 /// for the next level.
333 ///
334 /// This depends on the current level's block size, number of blocks and
335 /// is_internal bit vector.
336 ///
337 /// @param text The input text.
338 /// @param level The level data of the previous level.
339 /// @return The level data of the next level.
340 [[nodiscard]] LevelData
341 generate_next_level(const std::vector<input_type>& text,
342 const LevelData& level) const {
343 const size_t block_size = level.block_size;
344 const size_t num_blocks = level.num_blocks;
345 const auto& is_internal = *level.is_internal;
346 const size_t next_block_size = block_size / this->tau_;
347
348 std::vector<size_type> new_block_starts;
349 new_block_starts.reserve(num_blocks * this->tau_);
350 for (size_t i = 0; i < num_blocks; ++i) {
351 if (!is_internal[i]) {
352 continue;
353 }
354
355 // We generate up to tau new blocks for each internal block,
356 // excluding blocks that start past the end of the text
357 const auto parent_block_start = (*level.block_starts)[i];
358 for (size_t j = 0, current_block_start = parent_block_start;
359 j < static_cast<size_t>(this->tau_) &&
360 current_block_start < text.size();
361 ++j, current_block_start += next_block_size) {
362 new_block_starts.push_back(current_block_start);
363 }
364 }
365
366 LevelData next_level(level.level_index + 1,
367 next_block_size,
368 new_block_starts.size());
369 next_level.block_starts =
370 std::make_unique<std::vector<size_type>>(std::move(new_block_starts));
371 return next_level;
372 }

374 ///
375 /// @brief Takes a vector of levels and fills the block tree fields with them.
376 ///
377 /// @param[in] levels A vector containing data for each level, with the first
378 /// entry corresponding to the topmost level.
379 ///
380 void make_tree(const std::vector<input_type>& text,
381 std::vector<LevelData>& levels,
382 int64_t padding) {
383 const bool is_padded = padding > 0;
384
385 // Count the current number of internal blocks per level
386 std::vector<size_type> new_num_internal(levels.size(), 0);
387 for (size_t level = 0; level < levels.size(); level++) {
388 for (size_t block = 0; block < levels[level].is_internal->size();
389 block++) {
390 if ((*levels[level].is_internal)[block]) {
391 new_num_internal[level]++;
392 }
393 }
394 }
395
396 // Create first level
397 bool found_back_block = levels[0].is_internal->size() >
398 static_cast<size_t>(new_num_internal[0]) ||
399 !this->CUT_FIRST_LEVELS;
400 LevelData& top_level = levels.front();
401 if (found_back_block) {
402 const size_t n = top_level.num_blocks;
403 const size_t num_internal = new_num_internal[0];
404 auto pointers = new sdsl::int_vector<>(n - num_internal, 0);
405 auto offsets = new sdsl::int_vector<>(n - num_internal, 0);
406 size_t num_back_blocks = 0;
407 for (size_t i = 0; i < n; i++) {
408 // if a back block is found, add its pointer and offset
409 if (!(*top_level.is_internal)[i]) {
410 (*pointers)[num_back_blocks] = (*top_level.pointers)[i];
411 (*offsets)[num_back_blocks] = (*top_level.offsets)[i];
412 num_back_blocks++;
413 }
414 }
415 sdsl::util::bit_compress(*pointers);
416 sdsl::util::bit_compress(*offsets);
417 this->block_tree_types_.push_back(top_level.is_internal.release());
418 this->block_tree_types_rs_.push_back(
419 new Rank(*this->block_tree_types_.back()));
420 this->block_tree_pointers_.push_back(pointers);
421 this->block_tree_offsets_.push_back(offsets);
422 this->block_size_lvl_.push_back(top_level.block_size);
423 }
424 top_level.pointers.reset();
425 top_level.offsets.reset();
426 top_level.counters.reset();
427
428 // Add level data to the tree
429 for (size_t level_index = 1; level_index < levels.size(); level_index++) {
430 LevelData& level = levels[level_index];
431 LevelData& previous_level = levels[level_index - 1];
432 found_back_block |= static_cast<size_t>(new_num_internal[level_index]) <
433 levels[level_index].is_internal->size();
434 if (!found_back_block && level_index < levels.size() - 1) {
435 if (level_index < levels.size() - 1) {
436 level.is_internal.reset();
437 }
438 level.is_internal_rank.reset();
439 level.pointers.reset();
440 level.offsets.reset();
441 level.counters.reset();
442 previous_level.block_starts.reset();
443 continue;
444 }
445
446 make_tree_level(levels,
447 new_num_internal,
448 level_index,
449 is_padded,
450 text.size());
451
452 // We don't need these anymore
453 if (level_index < levels.size() - 1) {
454 level.is_internal.reset();
455 }
456 level.is_internal_rank.reset();
457 level.pointers.reset();
458 level.offsets.reset();
459 level.counters.reset();
460 previous_level.block_starts.reset();
461 }
462
463 this->leaf_size = levels.back().block_size / this->tau_;
464 // Construct the leaf string
465 int64_t leaf_count = 0;
466 auto& last_is_internal = *levels.back().is_internal;
467 std::vector<size_type>& last_block_starts = *levels.back().block_starts;
468 for (size_t block = 0; block < last_is_internal.size(); block++) {
469 if (!last_is_internal[block]) {
470 continue;
471 }
472 const size_type block_start = last_block_starts[block];
473 // For every leaf on the last level, we have tau leaf blocks
474 leaf_count += this->tau_;
475 // Iterate through all characters in this child and
476 // add them to the leaf string
477 for (size_t b = 0; b < static_cast<size_t>(this->leaf_size * this->tau_);
478 b++) {
479 if (static_cast<size_t>(block_start + b) < text.size()) {
480 this->leaves_.push_back(text[block_start + b]);
481 } else {
482 this->leaves_.push_back(0);
483 }
484 }
485 }
486 this->amount_of_leaves = leaf_count;
487 this->compress_leaves();
488 }

490 /// @brief Generates a level and adds the relevant data to the block tree.
491 ///
492 /// @param levels The vector of levels of the tree.
493 /// @param level_index The index of the level to generate. This must be
494 /// strictly greater than 0.
495 /// @param is_padded Whether there is padding in the last block of the tree
496 void make_tree_level(std::vector<LevelData>& levels,
497 const std::vector<size_type>& new_num_internal,
498 const size_t level_index,
499 const bool is_padded,
500 const size_t text_len) {
501 LevelData& previous_level = levels[level_index - 1];
502 LevelData& level = levels[level_index];
503
504 size_type new_size =
505 (new_num_internal[level_index - 1] - is_padded) * this->tau_;
506 // Determine the number of children the last block generated
507 if (is_padded) {
508 const size_type last_block_parent_start =
509 previous_level.block_starts->back();
510 const size_type block_size = level.block_size;
511 new_size +=
512 internal::sharded::ceil_div(text_len - last_block_parent_start,
513 block_size);
514 }
515 previous_level.block_starts.reset();
516 const size_type num_internal = new_num_internal[level_index];
517
518 // Allocate new vectors for the tree
519 auto* is_internal = new BitVector(new_size);
520 auto* pointers = new sdsl::int_vector<>(new_size - num_internal, 0);
521 auto* offsets = new sdsl::int_vector<>(new_size - num_internal, 0);
522
523 // Number of non-pruned blocks before the current block
524 size_type num_non_pruned = 0;
525 // Number of back blocks before the current block
526 size_type num_back_blocks = 0;
527 // Number of pruned blocks before the current block
528 size_type num_pruned = 0;
529
530 // We will reuse the allocated memory of the pointers vector to store
531 // the number of pruned blocks before the block.
532 // The invariant is that all values up to i are overwritten while all
533 // values starting after i will still be valid pointers
534 // This contains the number of pruned blocks before the block i
535 std::vector<size_type>& prefix_pruned_blocks = *level.pointers;
536 for (size_type i = 0; i < level.num_blocks; i++) {
537 const size_type ptr = (*level.pointers)[i];
538 prefix_pruned_blocks[i] = num_pruned;
539
540 // If the current block is not pruned, add it to the new tree
541 if (ptr == internal::sharded::PRUNED) {
542 num_pruned++;
543 continue;
544 }
545
546 // Add it to the is_internal bit vector
547 const bool block_is_internal = (*level.is_internal)[i];
548 (*is_internal)[num_non_pruned] = block_is_internal;
549 num_non_pruned++;
550
551 if (block_is_internal) {
552 continue;
553 }
554
555 // If it is a back block, add its pointer and offset
556 const size_type offset = (*level.offsets)[i];
557
558 (*pointers)[num_back_blocks] = ptr - prefix_pruned_blocks[ptr];
559 (*offsets)[num_back_blocks] = offset;
560 num_back_blocks++;
561 }
562
563 sdsl::util::bit_compress(*pointers);
564 sdsl::util::bit_compress(*offsets);
565 this->block_tree_types_.push_back(is_internal);
566 this->block_tree_types_rs_.push_back(new Rank(*is_internal));
567 this->block_tree_pointers_.push_back(pointers);
568 this->block_tree_offsets_.push_back(offsets);
569 this->block_size_lvl_.push_back(level.block_size);
570 }

572 /// @brief Prunes the tree of unnecessary nodes.
573 /// @param levels The levels of the tre represented as a vector of levels.
574 void prune(std::vector<LevelData>& levels) {
575 // We need to traverse the block tree in post order,
576 // handling children from right to left.
577 for (int block_index = levels[0].num_blocks - 1; block_index >= 0;
578 --block_index) {
579 prune_block(levels, 0, block_index);
580 }
581 }

583 /// @brief Prunes a block and its descendants of unnecessary internal nodes.
584 /// @param levels The WIP levels of the tree.
585 /// @param level_index The level of the block to prune.
586 /// @param block_index The index of the block to prune.
587 /// @return Whether this block is/stays internal after the pruning process
588 bool prune_block(std::vector<LevelData>& levels,
589 const size_t level_index,
590 const size_t block_index) {
591 LevelData& level = levels[level_index];
592 BitVector& is_internal = *level.is_internal;
593
594 // If the current block is a back block already, there is nothing to prune
595 if (!is_internal[block_index]) {
596 return false;
597 }
598
599 const size_type first_child =
600 level.is_internal_rank->rank1(block_index) * this->tau_;
601
602 bool has_internal_children = false;
603
604 // On the last level, all blocks just have leaves as children,
605 // none of which can be pointed to. So only recurse, if we are not on the
606 // last level.
607 if (level_index < levels.size() - 1) {
608 const size_type last_child =
609 std::min<size_type>(first_child + this->tau_ - 1,
610 levels[level_index + 1].is_internal->size() - 1);
611 // Iterate through children in reverse
612 for (size_type child = last_child; child >= first_child; --child) {
613 has_internal_children |= prune_block(levels, level_index + 1, child);
614 }
615 }
616
617 // If any of the children is internal, this block stays internal as well
618 if (has_internal_children) {
619 return true;
620 }
621
622 const size_type pointer = (*level.pointers)[block_index];
623 const size_type offset = (*level.offsets)[block_index];
624 const size_type counter = (*level.counters)[block_index];
625 // If there is no earlier occurrence or there are blocks pointing to this,
626 // then this must stay internal
627 if (pointer == internal::sharded::NO_EARLIER_OCC || counter > 0) {
628 return true;
629 }
630
631 // Now we know that there is an earlier occurrence,
632 // and nothing is pointing here.
633 // We will make this block here into a back block...
634 is_internal[block_index] = false;
635 (*level.counters)[pointer] += 1;
636 (*level.counters)[pointer + 1] += offset > 0;
637
638 if (level_index == levels.size() - 1) {
639 return false;
640 }
641
642 // ...and mark the children as pruned
643 LevelData& child_level = levels[level_index + 1];
644 const size_type last_child =
645 std::min<size_type>(first_child + this->tau_ - 1,
646 child_level.is_internal->size() - 1);
647
648 for (size_type child = last_child; child >= first_child; --child) {
649 const size_type child_pointer = (*child_level.pointers)[child];
650 const size_type child_offset = (*child_level.offsets)[child];
651 // Decrement the counter of where the child points
652 (*child_level.counters)[child_pointer] -= 1;
653 (*child_level.counters)[child_pointer + 1] -= child_offset > 0;
654 // Mark the child as pruned
655 (*child_level.pointers)[child] = internal::sharded::PRUNED;
656 }
657
658 return false;
659 }
