41/// @brief A hash map that must be used by multiple threads, each thread having
42/// only having write access to a certain segment of the input space.
43/// @tparam K The type of the keys in the hash map.
44/// @tparam V The type of the values in the hash map.
45/// @tparam SeqHashMapType The type of the hash map used internally.
46/// This should be compatible with std::unordered_map.
47/// @tparam UpdateFn The update function deciding how to insert or update values
48/// in the map.
49template <std::copy_constructible K,
50 std::move_constructible V,
51 template <typename, typename> typename SeqHashMapType =
52 std::unordered_map,
53 UpdateFunction<K, V> UpdateFn = pasta::Overwrite<K, V>>
54 requires std::movable<typename UpdateFn::InputValue>
55class SyncShardedMap {
56 /// The sequential backing hash map type
57 using SeqHashMap = SeqHashMapType<K, V>;
58 /// The sequential hash map's hasher
59 using Hasher = typename SeqHashMap::hasher;
60 /// The type used for updates
61 using InputValue = typename UpdateFn::InputValue;
62
63 /// The actual pair of key and value stored in the map
64 using StoredValue = std::pair<K, InputValue>;
65
66 using Queue = std::span<StoredValue>;
67
68 /// The memory order namespace from the standard library
69 using mem = std::memory_order;
70
71 /// @brief The number of threads operating on this map.
72 const size_t thread_count_;
73 /// @brief Contains a hash map for each thread
74 std::vector<SeqHashMap> map_;
75 /// @brief Contains a task queue for each thread, holding insert
76 /// operations for each thread.
77 std::vector<Queue> task_queue_;
78 /// @brief Contains the number of tasks in each thread's queue.
79 std::span<std::atomic_size_t> task_count_;
80 /// @brief Contains the number of threads currently handling their queues.
81 /// This is used 1. signal to other threads that they should handle their
82 /// queue, and 2. to keep track of whether all threads have handled their
83 /// queues.
84 std::atomic_size_t threads_handling_queue_;
85
86 const size_t queue_capacity_;
87
88#ifdef BT_INSTRUMENT
89 std::atomic_size_t num_cycles_;
90 std::function<void()> FN = [this]() noexcept {
91 this->num_cycles_.fetch_add(1, mem::acq_rel);
92 };
93#else
94 constexpr static std::invocable auto FN = []() noexcept {
95 };
96#endif
97
98 std::barrier<decltype(FN)> barrier_;
99
100public:
101 std::atomic_size_t num_updates_;
102 std::atomic_size_t num_inserts_;
103 //
104 /// @brief Creates a new sharded map.
105 ///
106 /// @param thread_count The exact number of threads working on this map.
107 /// @param queue_capacity The maximum amount of tasks allowed in each queue.
108 /// @param map_capacity The initial capacity for each thread's local hash map.
109 ///
110 SyncShardedMap(size_t thread_count, size_t queue_capacity)
111 : thread_count_(thread_count),
112 map_(),
113 task_queue_(),
114 task_count_(),
115 threads_handling_queue_(0),
116 queue_capacity_(queue_capacity),
117 barrier_(static_cast<ptrdiff_t>(thread_count), FN),
118 num_updates_(0),
119 num_inserts_(0) {
120 map_.reserve(thread_count);
121 task_queue_.reserve(thread_count);
122 task_count_ =
123 std::span<std::atomic_size_t>(new std::atomic_size_t[thread_count],
124 thread_count);
125 for (size_t i = 0; i < thread_count; i++) {
126 map_.emplace_back();
127 task_queue_.emplace_back(new StoredValue[queue_capacity], queue_capacity);
128 task_count_[i] = 0;
129 }
130 }
131
132 ~SyncShardedMap() {
133 delete[] task_count_.data();
134 for (auto& queue : task_queue_) {
135 delete[] queue.data();
136 }
137 }

167 /// @brief Inserts or updates a new value in the map, depending on whether
168 /// @param k The key to insert or update a value for.
169 /// @param in_value The value with which to insert or update.
170 inline void insert_or_update_direct(const K& k, InputValue&& in_value) {
171 auto res = map_.find(k);
172 if (res == map_.end()) {
173 // If the value does not exist, insert it
174 K key = k;
175 V initial = UpdateFn::init(key, std::move(in_value));
176 map_.emplace(key, std::move(initial));
177#ifdef BT_INSTRUMENT
178 sharded_map_.num_inserts_.fetch_add(1, mem::acq_rel);
179#endif
180 } else {
181 // Otherwise, update it.
182 V& val = res->second;
183 UpdateFn::update(k, val, std::move(in_value));
184#ifdef BT_INSTRUMENT
185 sharded_map_.num_updates_.fetch_add(1, mem::acq_rel);
186#endif
187 }
188 }
189
190 void handle_queue_sync(const bool make_others_wait = true) {
191 if (make_others_wait) {
192 // If this value is >0 then other threads will also handle their queue
193 // when trying to insert
194 sharded_map_.threads_handling_queue_.fetch_add(1, mem::acq_rel);
195 }
196#ifdef BT_INSTRUMENT
197 auto now = std::chrono::high_resolution_clock::now();
198#endif
199 sharded_map_.barrier_.arrive_and_wait();
200#ifdef BT_INSTRUMENT
201 size_t ns_count = std::chrono::duration_cast<std::chrono::nanoseconds>(
202 std::chrono::high_resolution_clock::now() - now)
203 .count();
204 start_idle_ns_.add(ns_count);
205
206 now = std::chrono::high_resolution_clock::now();
207#endif
208 handle_queue();
209#ifdef BT_INSTRUMENT
210 ns_count = std::chrono::duration_cast<std::chrono::nanoseconds>(
211 std::chrono::high_resolution_clock::now() - now)
212 .count();
213 handle_queue_ns_.add(ns_count);
214
215 now = std::chrono::high_resolution_clock::now();
216#endif
217 if (make_others_wait) {
218 sharded_map_.threads_handling_queue_.fetch_sub(1, mem::acq_rel);
219 }
220 sharded_map_.barrier_.arrive_and_wait();
221#ifdef BT_INSTRUMENT
222 ns_count = std::chrono::duration_cast<std::chrono::nanoseconds>(
223 std::chrono::high_resolution_clock::now() - now)
224 .count();
225 finish_idle_ns_.add(ns_count);
226#endif
227 }

229 /// @brief Handles this thread's queue, inserting or updating all values in
230 /// its queue, waiting for other threads to be
231 /// done with their handle_queue call.
232 void handle_queue() {
233 const size_t num_tasks_raw = task_count_.exchange(0, mem::acq_rel);
234 BT_ASSERT(num_tasks_raw <= sharded_map_.queue_capacity_);
235 const size_t num_tasks =
236 std::min(num_tasks_raw, sharded_map_.queue_capacity_);
237 if (num_tasks == 0) {
238 return;
239 }
240
241 // Handle all tasks in the queue
242 for (size_t i = 0; i < num_tasks; ++i) {
243 auto& entry = task_queue_[i];
244 insert_or_update_direct(entry.first, std::move(entry.second));
245 }
246 }
247
248 /// @brief Inserts or updates a new value in the map.
249 ///
250 /// If the value is inserted into the current thread's map,
251 /// it is inserted immediately. If not, then it is added to that thread's
252 /// queue. It will only be inserted into the map, once the thread comes
253 /// around to handle its queue using the handle_queue method.
254 ///
255 /// @param pair The key-value pair to insert or update.
256 void insert(StoredValue&& pair) {
257 if (sharded_map_.threads_handling_queue_.load(mem::acquire) > 0) {
258 handle_queue_sync();
259 }
260 const size_t hash = Hasher{}(pair.first);
261 const size_t target_thread_id =
262 internal::sharded::mix_select(hash) % sharded_map_.thread_count_;
263
264 // Otherwise enqueue the new value in the target thread
265 std::atomic_size_t& target_task_count =
266 sharded_map_.task_count_[target_thread_id];
267
268 size_t task_idx = target_task_count.fetch_add(1, mem::acq_rel);
269 // If the target queue is full, signal to the other threads, that they
270 // need to handle their queue and handle this thread's queue
271 if (task_idx >= sharded_map_.queue_capacity_) {
272 // Since we incremented that thread's task count, but didn't insert
273 // anything, we need to decrement it again so that it has the correct
274 // value
275 target_task_count.fetch_sub(1, mem::acq_rel);
276 handle_queue_sync();
277 // Since the queue was handled, the task count is now 0
278 insert(std::move(pair));
279 return;
280 }
281 // Insert the value into the queue
282 sharded_map_.task_queue_[target_thread_id][task_idx] = std::move(pair);
283 }

285 /// @brief Inserts or updates a new value in the map.
286 ///
287 /// If the value is inserted into the current thread's map,
288 /// it is inserted immediately. If not, then it is added to that thread's
289 /// queue. It will only be inserted into the map, once the thread comes
290 /// around to handle its queue using the handle_queue method.
291 ///
292 /// @param key The key of the value to insert.
293 /// @param value The value to associate with the key.
294 inline void insert(K& key, InputValue value) {
295 insert(StoredValue(key, value));
296 }

323 /// @brief Returns the number of key-value pairs in the map.
324 ///
325 /// Note, that this method calculates the size for each map separately and
326 /// is therefore not O(1).
327 /// @return The number of key-value pairs in the map.

329 size_t size = 0;

331 size += map.size();

336 [[maybe_unused]] Whereabouts where(const K& k) {
337 const size_t hash = Hasher{}(k);
338 const size_t target_thread_id =
339 internal::sharded::mix_select(hash) % thread_count_;
340 SeqHashMap& map = map_[target_thread_id];
341 typename SeqHashMap::iterator it = map.find(k);
342 if (it != map.end()) {
343 return IN_MAP;
344 }
345 Queue& queue = task_queue_[target_thread_id];
346 for (size_t i = 0; i < task_count_[target_thread_id]; ++i) {
347 if (queue[i].first == k) {
348 return IN_QUEUE;
349 }
350 }
351 return NOWHERE;
352 }

354 /// @brief Runs a method for each value in the map.
355 ///
356 /// The given function must take const references to a key and a value
357 /// respectively.
358 /// @param f The function or lambda to run for each value.
359 void for_each(std::invocable<const K&, const V&> auto f) const {
360 for (const SeqHashMap& map : map_) {
361 for (const auto& [k, v] : map) {
362 f(k, v);
363 }
364 }
365 }

367 typename SeqHashMap::iterator end() {
368 return map_.back().end();
369 }

371 typename SeqHashMap::iterator find(const K& key) {
372 const size_t hash = Hasher{}(key);
373 const size_t target_thread_id =
374 internal::sharded::mix_select(hash) % thread_count_;
375 SeqHashMap& map = map_[target_thread_id];
376 typename SeqHashMap::iterator it = map.find(key);
377 if (it == map.end()) {
378 return end();
379 }
380 return it;
381 }
