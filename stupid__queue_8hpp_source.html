36 bool enqueue(std::unique_ptr<T>&& elem) {
37 std::unique_lock<std::mutex> lock(enqueue_mtx_);
38 const bool can_insert =
39 enqueue_cv_.wait_for(lock, std::chrono::milliseconds(10), [this] {
40 return size() < capacity_;

42 if (!can_insert) {
43 return false;
44 }
45 assert(size() < capacity_);

56 queue_.pop_front();
57 }
58 enqueue_cv_.notify_one();
