30template <uint8_t exponent>
31static consteval uint128_t mersenne_prime() {
32 uint128_t res = 1;
33 for (size_t i = 0; i < exponent; i++) {
34 res <<= 1;
35 }
36 return res - 1;
37}

39///
40/// @brief A Rabin-Karp rolling hasher.
41///
42/// @tparam T The type of the characters in the text.
43/// @tparam size_type The type to use for indexing etc.
44/// @tparam mersenne_exponent If using a mersenne prime 2^p-1, then this should
45/// be p. If this is 0, a normal modulus operation will be used.
46/// Additionally, if this is != 0, then the prime_ attribute will be ignored
47/// and '(1 << mersenne_exponent) - 1' will be used instead.
48///
49template <class T, class size_type, uint8_t mersenne_exponent = 0>
50class MersenneRabinKarp {
51public:
52 /// The text being hashed
53 std::span<const T> text_;
54 uint128_t sigma_;
55 /// The start index of the currently hashed window
56 uint64_t init_;
57 /// The window size of this hasher
58 uint64_t length_;
59 /// A large prime used for modulus operations
60 uint128_t prime_;
61 /// The current hash value
62 uint128_t hash_;
63 uint128_t max_sigma_;
64
65 /// @brief Construct a new Rabin Karp hasher.
66 /// @param text The text to hash. (not just the window but the entire text)
67 /// @param sigma The alphabet size.
68 /// @param init The start index of the first hashed window in the text.
69 /// @param length The window size.
70 /// @param prime A large prime used for modulus operations
71 /// iff not using mersenne_exponent.
72 constexpr MersenneRabinKarp(const std::span<const T> text,
73 const uint64_t sigma,
74 const uint64_t init,
75 const uint64_t length,
76 const uint128_t prime)
77 : text_(text),
78 sigma_(sigma),
79 init_(init),
80 length_(length),
81 prime_(prime) {
82 max_sigma_ = 1;
83 uint128_t fp = 0;
84 uint128_t sigma_c = 1;
85 for (uint64_t i = init_; i < init_ + length_; i++) {
86 fp = fp * sigma;
87 fp = mersenneModulo(fp + text_[i]);
88 }
89 for (uint64_t i = 0; i < length_ - 1; i++) {
90 sigma_c = mersenneModulo(sigma_c * sigma_);
91 }
92 hash_ = fp;
93 max_sigma_ = sigma_c;
94 };
95
96 constexpr MersenneRabinKarp(const std::vector<T>& text,
97 const uint64_t sigma,
98 const uint64_t init,
99 const uint64_t length,
100 const uint128_t prime)
101 : MersenneRabinKarp(std::span(text), sigma, init, length, prime) {}
102
103 /// @brief Moves the hasher to the specified start index in the backing
104 /// vector.
105 void restart(const uint64_t index) {
106 if (index + length_ >= text_.size()) {
107 return;
108 }
109 init_ = index;
110 uint128_t fp = 0;
111 for (uint64_t i = init_; i < init_ + length_; i++) {
112 fp = fp * sigma_;
113 fp = mersenneModulo(fp + text_[i]);
114 }
115 hash_ = fp;
116 };
117
118 constexpr uint128_t mersenneModulo(uint128_t k) const {
119 if constexpr (mersenne_exponent == 0) {
120 return k % prime_;
121 } else {
122 constexpr uint128_t MERSENNE = mersenne_prime<mersenne_exponent>();
123 uint128_t i = (k & MERSENNE) + (k >> mersenne_exponent);
124 i -= (i >= MERSENNE) * MERSENNE;
125 return i;
126 }
127 };
128
129 /// @brief Retrieves the hash value at the hasher's current position.
130 /// @return A MersenneHash object representing the current hash value.
131 constexpr MersenneHash<T> current_hash() const {
132 return MersenneHash<T>(text_, hash_, init_, length_);
133 }
134
135 /// @brief Advances the hasher by one character.
136 constexpr void next() {
137 if (text_.size() <= init_ + length_) {
138 return;
139 }
140
141 uint128_t fp = hash_;
142 T out_char = text_[init_];
143 T in_char = text_[init_ + length_];
144 const uint128_t out_char_influence = mersenneModulo(out_char * max_sigma_);
145 // Conditionally add the prime, of the out_char_influence is too large
146 if constexpr (mersenne_exponent == 0) {
147 fp += prime_ * (out_char_influence > hash_) - out_char_influence;
148 } else {
149 fp += mersenne_prime<mersenne_exponent>() * (out_char_influence > hash_) -
150 out_char_influence;
151 }
152 fp *= sigma_;
153 fp += in_char;
154 fp = mersenneModulo(fp);
155 hash_ = fp;
156 init_++;
157 };
158};

160///
161/// @brief A Rabin-Karp rolling hasher for bitstrings.
162///
163/// @tparam size_type The type to use for indexing etc.
164/// @tparam mersenne_exponent If using a mersenne prime 2^p-1, then this should
165/// be p. If this is 0, a normal modulus operation will be used.
166/// Additionally, if this is != 0, then the prime_ attribute will be ignored
167/// and '(1 << mersenne_exponent) - 1' will be used instead.
168///
169template <typename size_type, uint8_t mersenne_exponent>
170class MersenneRabinKarp<bool, size_type, mersenne_exponent> {
171public:
172 /// The text being hashed
173 const pasta::BitVector& text_;
174 uint64_t init_;
175 /// The window size of this hasher
176 uint64_t length_;
177 /// A large prime used for modulus operations
178 uint128_t prime_;
179 /// The current hash value
180 uint128_t hash_;
181 uint128_t max_sigma_;
182
183 /// @brief Construct a new Rabin Karp hasher over a bitstring.
184 /// @param text The bitvector to hash. (not just the window but the entire
185 /// text)
186 /// @param init The start bit-index of the first hashed window in the text.
187 /// @param length The window size in bits.
188 /// @param prime A large prime used for modulus operations
189 /// iff not using mersenne_exponent.
190 constexpr MersenneRabinKarp(const pasta::BitVector& text,
191 const uint64_t init,
192 const uint64_t length,
193 const uint128_t prime)
194 : text_(text),
195 init_(init),
196 length_(length),
197 prime_(prime) {
198 max_sigma_ = 1;
199 uint128_t fp = 0;
200 uint128_t sigma_c = 1;
201 for (uint64_t i = init_; i < init_ + length_; i++) {
202 fp = mersenneModulo(2 * fp + get_bit(i));
203 }
204 for (uint64_t i = 0; i < length_ - 1; i++) {
205 sigma_c = mersenneModulo(2 * sigma_c);
206 }
207 hash_ = fp;
208 max_sigma_ = sigma_c;
209 }
210
211 /// @brief Moves the hasher to the specified start index in the backing
212 /// vector.
213 constexpr void restart(const uint64_t index) {
214 if (index + length_ >= text_.size()) {
215 return;
216 }
217 init_ = index;
218 uint128_t fp = 0;
219 for (uint64_t i = init_; i < init_ + length_; i++) {
220 fp = fp * 2;
221 fp = mersenneModulo(fp + get_bit(i));
222 }
223 hash_ = fp;
224 };
225
226 [[nodiscard]] constexpr uint128_t mersenneModulo(const uint128_t k) const {
227 if constexpr (mersenne_exponent == 0) {
228 return k % prime_;
229 } else {
230 constexpr uint128_t MERSENNE = mersenne_prime<mersenne_exponent>();
231 uint128_t i = (k & MERSENNE) + (k >> mersenne_exponent);
232 i -= (i >= MERSENNE) * MERSENNE;
233 return i;
234 }
235 };
236
237 /// @brief Retrieves the hash value at the hasher's current position.
238 /// @return A MersenneHash object representing the current hash value.
239 [[nodiscard]] constexpr MersenneHash<bool> current_hash() const {
240 return {text_, hash_, init_, length_};
241 }
242
243 /// @brief Advances the hasher by one character.
244 constexpr void next() {
245 if (text_.size() <= init_ + length_) {
246 return;
247 }
248
249 uint128_t fp = hash_;
250 const bool out_char = out_bit();
251 const bool in_char = in_bit();
252 const uint128_t out_char_influence = out_char * max_sigma_;
253 // Conditionally add the prime, of the out_char_influence is too large
254 if constexpr (mersenne_exponent == 0) {
255 fp += prime_ * (out_char_influence > hash_) - out_char_influence;
256 } else {
257 fp += mersenne_prime<mersenne_exponent>() * (out_char_influence > hash_) -
258 out_char_influence;
259 }
260 fp *= 2;
261 fp += in_char;
262 fp = mersenneModulo(fp);
263 hash_ = fp;
264 init_++;
265 };
266
267private:
268 [[nodiscard]] constexpr bool out_bit() const {
269 return get_bit(init_);
270 }
271
272 [[nodiscard]] constexpr bool in_bit() const {
273 const size_t idx = init_ + length_;
274 return get_bit(idx);
275 }
276
277 [[nodiscard]] constexpr bool get_bit(const size_t bit_index) const {
278 return text_[bit_index];
279 }
280};
